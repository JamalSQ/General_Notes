***************************change the new file (newssite.conf) AFTER*****************************
<VirtualHost *:80>
	ServerAdmin webmaster@newsite
	ServerName newssite.com
	ServerAlias www.newssite.com
	DocumentRoot /var/www/newssite
	ErrorLog ${APACHE_LOG_DIR}/error.log
	CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost>
