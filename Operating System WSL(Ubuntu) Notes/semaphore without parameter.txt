#include<unistd.h>
#include<stdio.h>
#include<pthread.h>
void *fun1();
void *fun2();

int main(){
        pthread_t th1,th2;
	int a=0,b=0;
        pthread_create(&th1,NULL,fun1,a);
        pthread_create(&th2,NULL,fun2,b);

//use join to for witing the termination of threads
        pthread_join(th1,NULL);
        pthread_join(th2,NULL);
}

void *fun1(int a){

for( a=0; a<5; a++){
printf("%d\t",a);
}

}

for( b=0; b<5; b++){
